请求基类：RequestBase

    /// <summary>

    /// APP请求基类

    /// </summary>

    public class RequestBase

    {

        /// <summary>

        /// 用户编号

        /// </summary>

        public int? UserId { get; set; }

 

        /// <summary>

        /// 签名密钥

        /// </summary>

        public string Sign { get; set; }

 

        /// <summary>

        ///  密钥Token

        /// </summary>

        public string Token { get; set; }

 

        /// <summary>

        /// 平台类型 1 Android 2 iPhone

        /// </summary>

        public int Platform { get; set; }

    }
